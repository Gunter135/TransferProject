Небольшая помарка this is not CMD script, it's VBScript

' ==========================================
' CONFIGURATION - EDIT THESE LINES
' ==========================================
Dim searchText : searchText = "orm"
Dim replaceText : replaceText = "ume"
Dim startFolder : startFolder = "." ' "." means current folder
Dim fileFilter : fileFilter = "*.txt,*.js,*.css,*.html,*.xml,*.json,*.bat,*.cmd"
' ==========================================

On Error Resume Next

Dim fso, rootFolder
Set fso = CreateObject("Scripting.FileSystemObject")

' Get the absolute path of the starting folder
If startFolder = "." Then
    Set rootFolder = fso.GetFolder(fso.GetAbsolutePathName("."))
Else
    Set rootFolder = fso.GetFolder(startFolder)
End If

WScript.Echo "Starting Find & Replace (VBScript)..."
WScript.Echo "Search: " & searchText
WScript.Echo "Replace: " & replaceText
WScript.Echo "Folder: " & rootFolder.Path
WScript.Echo "------------------------------------------"

Dim fileCount, modCount
fileCount = 0
modCount = 0

' Start the recursion
ProcessFolder rootFolder

WScript.Echo "------------------------------------------"
WScript.Echo "Done! Scanned: " & fileCount & ", Modified: " & modCount

Sub ProcessFolder(folder)
    Dim file, subFolder, filePath, content, newContent, ext, shouldProcess
    
    ' Process Files
    For Each file In folder.Files
        fileCount = fileCount + 1
        filePath = file.Path
        ext = LCase(fso.GetExtensionName(filePath))
        
        ' Check filter (simple check)
        shouldProcess = False
        If InStr(fileFilter, "*.*") > 0 Then
            shouldProcess = True
        Else
            Dim filterItem
            For Each filterItem In Split(fileFilter, ",")
                If LCase(file.Name) Like LCase(Trim(filterItem)) Then
                    shouldProcess = True
                    Exit For
                End If
            Next
        End If
        
        If shouldProcess Then
            On Error Resume Next
            ' Read File
            Dim ts
            Set ts = fso.OpenTextFile(filePath, 1, False, -1) ' -1 = TristateUseDefault
            If Err.Number = 0 Then
                content = ts.ReadAll
                ts.Close
                
                ' Check and Replace
                If InStr(content, searchText) > 0 Then
                    newContent = Replace(content, searchText, replaceText, 1, -1, 0)
                    
                    ' Write Back
                    Set ts = fso.OpenTextFile(filePath, 2, False, -1)
                    ts.Write newContent
                    ts.Close
                    modCount = modCount + 1
                    WScript.Echo "[MODIFIED] " & filePath
                End If
            End If
            On Error Goto 0
        End If
    Next
    
    ' Process Subfolders
    For Each subFolder In folder.SubFolders
        ' Skip system hidden folders often causing issues
        If (subFolder.Attributes And 2) = 0 Then 
            ProcessFolder subFolder
        End If
    Next
End Sub
